---
title: Getting Started
description: Get up and running with Locallytics in minutes
---

## 1. Installation

Install the package:

```bash tab="npm"
npm install locallytics
```

```bash tab="pnpm"
pnpm install locallytics
```

```bash tab="yarn"
yarn add locallytics
```

## 2. Set up client

Create an API route for handling analytics and initializing the locallytics instance:

```tsx title=app/api/analytics/route.ts
import { locallytics } from "locallytics";

export const analytics = await locallytics({
  database: process.env.DATABASE_URL!,
});

export const { GET, POST } = analytics;
```

## 3. Generate database schema

Generate the database schema:

```bash tab="npm"
npx locallytics generate
```

```bash tab="pnpm"
pnpm locallytics generate
```

```bash tab="yarn"
yarn locallytics generate
```

## 4. Run migrations

Apply the database migrations:

```bash tab="npm"
npx locallytics migrate
```

```bash tab="pnpm"
pnpm locallytics migrate
```

```bash tab="yarn"
yarn locallytics migrate
```

## 5. Add AnalyticsGrabber to your app's root layout

```tsx title=app/layout.tsx
import { AnalyticsGrabber } from "locallytics";

export default function RootLayout({ children }) {
  return (
    <html>
      <body>
        {children}
        <AnalyticsGrabber />
      </body>
    </html>
  );
}
```

## 6. View analytics

Create a page to display your analytics data:

```tsx title=app/analytics/page.tsx
import { AnalyticsJSON } from "locallytics";

export default async function AnalyticsPage() {
  const data = await AnalyticsJSON();

  return (
    <div>
      <h1>Analytics</h1>
      <pre>{JSON.stringify(data, null, 2)}</pre>
    </div>
  );
}
```
